# Extensible metasyntax: quote words

    # Leading space after < means "quote words" array literal

    my $str = "ruby";

    if $str ~~ m/ < perl python ruby > / {
        # same as: m/ [ 'perl' | 'python' | 'ruby' ] /
    	say "Matched '$/'";  # ruby
    }
