# Operators: Doubling down on Unix

    my @input = 1..20;
    my (@evens, @odds);

    # Feed operator, strictly lazy

    @input ==> grep { $_ % 2 == 0 } ==> @evens;  # evens
    say @evens;

    # Feed operators have the notion of "direction"

    @odds <== grep { $_ % 2 != 0 } <== @input;
    say @odds;
