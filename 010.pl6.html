# Interpolation

    my $name = "Gabriel";
    say "Hello, $name";

    my @list = <one two three four>;  # qw/one two three four/
    say "@list";                      # literal, no interpolation
    say "@list[]";                    # zen slice
    say "{@list}";                    # interpolates

    my %hash = ( 1 => 'one', 2 => 'two', );
    say "%hash";                      # literal, no interpolation
    say "%hash{}";                    # interpolates
    say "{%hash}";                    # also interpolates
