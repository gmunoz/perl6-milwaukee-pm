# Inheritance
    class Person {
    	has $.name is rw;

    	method reset {
    		$.name = "";
    	}
    }

    class Monger is Person {
    	has $.fav_lang is rw;

    	method reset {
    		$.fav_lang = "perl";
    		nextsame;
    	}
    }

    my $monger = Monger.new(name => "Gabriel", fav_lang => "perl6");
    say $monger.fav_lang;
    $monger.reset;
    say $monger.fav_lang;
